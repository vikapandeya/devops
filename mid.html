Jenkins Declarative Pipeline

// Jenkinsfile
pipeline {
  agent { label 'docker' }

  environment {
    IMAGE = "myapp:${env.BUILD_NUMBER}"
    REGISTRY = "registry.company.com"
  }

  stages {
    stage('Checkout') {
      steps { checkout scm }
    }

    stage('Test') {
      parallel {                          // run tests in parallel
        stage('Unit Tests') {
          steps { sh 'npm run test:unit' }
        }
        stage('Lint') {
          steps { sh 'npm run lint' }
        }
      }
    }

    stage('Build') {
      steps {
        sh "docker build -t ${REGISTRY}/${IMAGE} ."
        withCredentials([usernamePassword(credentialsId: 'registry', usernameVariable: 'USER', passwordVariable: 'PASS')]) {
          sh "docker login ${REGISTRY} -u ${USER} -p ${PASS}"
          sh "docker push ${REGISTRY}/${IMAGE}"
        }
      }
    }

    stage('Deploy Staging') {
      steps {
        sh "kubectl set image deployment/app app=${REGISTRY}/${IMAGE} -n staging"
        sh "kubectl rollout status deployment/app -n staging"
      }
    }

    stage('Smoke Test') {
      steps {
        sh "curl -f https://staging.app.com/health || exit 1"
      }
    }

    stage('Deploy Production') {
      when { branch 'main' }
      input { message 'Deploy to production?' }
      steps {
        sh "kubectl set image deployment/app app=${REGISTRY}/${IMAGE} -n production"
        sh "kubectl rollout status deployment/app -n production"
      }
    }
  }

  post {
    failure { slackSend color: 'red', message: "FAILED: ${env.JOB_NAME} #${env.BUILD_NUMBER}" }
    success { slackSend color: 'green', message: "SUCCESS: ${env.JOB_NAME} #${env.BUILD_NUMBER}" }
  }
}

name: Build and Deploy

  push:
    branches: [main, staging]

jobs:
  build:
    runs-on: ubuntu-latest

      image-tag: ${{ steps.meta.outputs.tags }}

          registry: ghcr.io

          push: true

  deploy-staging:

    environment: staging
    runs-on: ubuntu-latest

      - uses: azure/k8s-deploy@v4

          namespace: staging
          images: ${{ needs.build.outputs.image-tag }}
          manifests: k8s/staging/

  deploy-prod:
    needs: build
    if: github.ref == 'refs/heads/main'
    environment: production      # requires manual approval in GitHub
    runs-on: ubuntu-latest
    steps:
      - uses: azure/k8s-deploy@v4
        with:
          namespace: production
          images: ${{ needs.build.outputs.image-tag }}

Multi-Stage Builds — Slim Production Images

# ── Build stage (full tools) ──
FROM node:20 AS builder
WORKDIR /app
COPY package*.json ./
RUN npm ci                      # install ALL deps including dev

RUN npm run build               # compile TypeScript → dist/

# ── Production stage (minimal) ──
FROM node:20-alpine AS production
WORKDIR /app
RUN addgroup -S appgroup && adduser -S appuser -G appgroup
COPY package*.json ./
RUN npm ci --only=production    # only prod deps
COPY --from=builder /app/dist ./dist   # copy compiled output only

USER appuser                    # never run as root!
EXPOSE 3000
CMD ["node", "dist/server.js"]

# Result: 1.2GB → 120MB image

  name: webapp
  namespace: production

  replicas: 3
  selector:
    matchLabels:
      app: webapp
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1           # max extra pods during update
      maxUnavailable: 0     # zero downtime
  template:
    metadata:
      labels:
        app: webapp
    spec:
      containers:
      - name: webapp
        image: myapp:1.0
        ports:
        - containerPort: 3000
        resources:
          requests: { cpu: 100m, memory: 128Mi }
          limits:   { cpu: 500m, memory: 512Mi }
        readinessProbe:
          httpGet: { path: /health, port: 3000 }
          initialDelaySeconds: 10
          periodSeconds: 5
        livenessProbe:
          httpGet: { path: /health, port: 3000 }
          initialDelaySeconds: 30
          failureThreshold: 3
        env:
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef: { name: db-secret, key: password }

# deploy-app.yml
---
- name: Deploy Application
  hosts: webservers                    # targets from inventory
  become: yes                          # sudo
  vars:
    app_version: "{{ version | default('latest') }}"
    app_dir: /opt/myapp

  pre_tasks:
    - name: Update apt cache
      apt: update_cache=yes cache_valid_time=3600
      when: ansible_os_family == "Debian"

  tasks:
    - name: Create app directory
      file:
        path: "{{ app_dir }}"
        state: directory
        owner: www-data
        mode: '0755'

    - name: Pull Docker image
      docker_image:
        name: "myregistry/myapp:{{ app_version }}"
        source: pull

    - name: Deploy container
      docker_container:
        name: myapp
        image: "myregistry/myapp:{{ app_version }}"
        state: started
        restart_policy: unless-stopped
        ports:
          - "3000:3000"
        env:
          DB_HOST: "{{ db_host }}"
          DB_PASS: "{{ vault_db_password }}"   # from Ansible Vault

    - name: Check app health
      uri:
        url: "http://localhost:3000/health"
        status_code: 200
      retries: 5
      delay: 10

  handlers:
    - name: reload nginx
      service: name=nginx state=reloaded

Interview Prep — Mid Level

Interview⏱ ~30 min
Explain Kubernetes resource requests vs limits ▼
Requests: The guaranteed minimum resources a container needs — used by the scheduler to decide which node to place the pod on. Limits: The maximum resources a container can use — if it exceeds CPU limit, it's throttled; if it exceeds memory limit, it's OOM-killed. Best practice: always set both. Set requests to the 50th percentile of actual usage and limits to 2-3x requests to handle spikes. Never set CPU requests too high as it leads to pods pending if nodes don't have that much free.
How does Terraform handle state and why is remote state important? ▼
Terraform stores the mapping between your HCL configuration and real infrastructure in a state file (terraform.tfstate). This allows it to know what exists and what needs to change. Remote state (in S3, Terraform Cloud) is critical for teams because: 1) Multiple engineers can work without conflicts — DynamoDB table provides locking to prevent simultaneous applies. 2) State is never stored in local files that could be lost or contain secrets. 3) State can be read as a data source by other Terraform configurations. Always enable encryption for the S3 bucket storing state.
What is the difference between liveness and readiness probes? ▼
Liveness probe: Determines if the container is running. If it fails, Kubernetes restarts the container. Use when the app can deadlock without crashing. Readiness probe: Determines if the container is ready to serve traffic. If it fails, the pod is removed from the Service endpoints (no traffic sent) but NOT restarted. Use during startup, DB connection warmup, or when processing a batch job. Tip: always set initialDelaySeconds to prevent probes from failing before the app starts up.
How do you implement zero-downtime deployments in Kubernetes? ▼
Multiple strategies: 1) RollingUpdate (default): Set maxUnavailable: 0 and maxSurge: 1 — never removes an old pod until a new one passes its readiness probe. 2) Proper readiness probes: No pod receives traffic until it's actually ready. 3) preStop hooks: Allow app to finish in-flight requests during graceful shutdown with terminationGracePeriodSeconds. 4) PodDisruptionBudgets: Ensure minimum available pods during voluntary disruptions (node drains). 5) Ingress connection draining: Load balancer stops sending new requests to pods being terminated.
Explain the Terraform plan/apply lifecycle ▼
terraform init: Downloads provider plugins, initializes backend (remote state). terraform plan: Compares desired state (HCL) with current state (state file) and actual infrastructure, shows what will be created/changed/destroyed — no changes made. terraform apply: Executes the plan, calls provider APIs to create/modify/delete resources, updates state file. terraform destroy: Plans and applies deletion of all managed resources. In CI/CD: run terraform plan -out=tfplan in PR, then terraform apply tfplan on merge to main.
How do you troubleshoot a pod stuck in CrashLoopBackOff? ▼
Step-by-step: 1) kubectl describe pod <pod> — check Events section for OOMKilled, image pull errors, or failed probes. 2) kubectl logs <pod> --previous — see logs from the last crash. 3) Check resource limits — OOMKilled means memory limit too low. 4) Verify image exists and tag is correct. 5) Check startup command — kubectl exec into a running similar container to test. 6) Check liveness probe settings — may be firing too early. 7) Validate environment variables and secrets are correctly mounted. CrashLoopBackOff itself just means the container keeps crashing; the cause is always in the logs or events.
